         from numpy.linalg import norm as euclidean
         from numpy import array as array
         def euclideanDistance(self,a):
            c=[]
            res=[]
            z=0
            while(len(a)>z):
                analysis=a.copy()
                del analysis[z]
                for i in analysis:
                    res.append(euclidean(array(a[z])-array(i)))
                c.append(tuple( [ a[z],analysis[res.index(min(res))]]))
                res.clear()
                z+=1
            return c
